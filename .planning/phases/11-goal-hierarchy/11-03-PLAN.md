    Follow the exact pattern from scheduler/planning.ts:
    ```typescript
    async function awaitResponse(dm: DMChannel, authorId: string, timeoutMs = 5 * 60 * 1000): Promise<string | null> {
      try {
        const collected = await dm.awaitMessages({
          filter: (msg: Message) => msg.author.id === authorId,
          max: 1,
          time: timeoutMs,
          errors: ['time'],
        });
        return collected.first()?.content ?? null;
      } catch {
        return null;
      }
    }
    ```
