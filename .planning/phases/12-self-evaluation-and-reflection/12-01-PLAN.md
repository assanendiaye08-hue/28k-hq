   model Reflection {
     id        String         @id @default(cuid())
     memberId  String
     type      ReflectionType // DAILY, WEEKLY, MONTHLY
     question  String         // The AI-generated question asked
     response  String         // Member's natural language response (encrypted)
     insights  String?        // AI-extracted key insight (1-2 sentences)
     createdAt DateTime       @default(now())

     member Member @relation(fields: [memberId], references: [id], onDelete: Cascade)

     @@index([memberId, createdAt])
     @@index([memberId, type])
   }

   enum ReflectionType {
     DAILY
     WEEKLY
     MONTHLY
   }
