    **settings-persistence.ts** (follow exact pattern from timer-persistence.ts):
    ```
    import { load } from '@tauri-apps/plugin-store';

    export interface AppSettings {
      autoUpdateEnabled: boolean;
      autoStartEnabled: boolean;
    }

    const DEFAULTS: AppSettings = {
      autoUpdateEnabled: false,
      autoStartEnabled: false,
    };

    export async function loadSettings(): Promise<AppSettings> {
      const store = await load('settings.json', { defaults: {} });
      const saved = await store.get<AppSettings>('settings');
      return saved ?? DEFAULTS;
    }

    export async function saveSettings(settings: AppSettings): Promise<void> {
      const store = await load('settings.json', { defaults: {} });
      await store.set('settings', settings);
      await store.save();
    }
    ```

    **updater.ts**:
    ```
    import { check } from '@tauri-apps/plugin-updater';
    import { relaunch } from '@tauri-apps/plugin-process';

    export async function checkForUpdate(): Promise<{ available: boolean; version?: string }> {
      try {
        const update = await check();
        if (!update) return { available: false };
        return { available: true, version: update.version };
      } catch {
        return { available: false };
      }
    }

    export async function downloadAndInstallUpdate(): Promise<void> {
      const update = await check();
      if (!update) return;
      await update.downloadAndInstall();
      await relaunch();
    }
    ```

    **settings-store.ts** (Zustand store):
    - State: `autoUpdateEnabled: boolean`, `autoStartEnabled: boolean`, `updateAvailable: { version: string } | null`, `isCheckingUpdate: boolean`, `isLoaded: boolean`
    - Actions:
      - `loadFromDisk()`: calls `loadSettings()`, also calls `isEnabled()` from `@tauri-apps/plugin-autostart` to sync OS autostart state, sets isLoaded=true
      - `setAutoUpdate(enabled: boolean)`: updates state, calls `saveSettings()`, if enabling then calls `checkForUpdate()` and sets updateAvailable if found
      - `setAutoStart(enabled: boolean)`: calls `enable()` or `disable()` from `@tauri-apps/plugin-autostart`, updates state, calls `saveSettings()`
      - `checkUpdate()`: sets isCheckingUpdate=true, calls `checkForUpdate()`, updates updateAvailable, sets isCheckingUpdate=false
      - `installUpdate()`: calls `downloadAndInstallUpdate()`

    Important: autostart reads OS state via `isEnabled()` as source of truth, not just the persisted setting. This avoids drift between persisted value and actual OS registration.
